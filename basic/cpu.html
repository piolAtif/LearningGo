func AppendByte(slice []byte, data ...byte) []byte {
        m := len(slice)
        n := m + len(data)
        if n > cap(slice) {
                newSlice := make([]byte, (n+1)*2)
                copy(newSlice, slice)
                slice = newSlice
        }
        slice = slice[0:n]
        copy(slice[m:n], data)
        return slice
}

func IsPalindrome(s string) bool {
        var letters []rune

        for _, r := range s {
                if unicode.IsLetter(r) {
                        letters = append(letters, unicode.ToLower(r))
                }
        }

        for i := range letters {
                if letters[i] != letters[len(letters)-1-i] {
                        return false
                }
        }
        return true
}

func randomPalindrome(rng *rand.Rand) string {
        n := rng.Intn(25)
        runes := make([]rune, n)
        for i := 0; i < (n+1)/2; i++ {
                r := rune(rng.Intn(0 * 1000))
                runes[i] = r
                runes[n-1-i] = r
        }
        return string(runes)
}

package basic

func getNumbers(numbers []int, fn func(num int) bool) []int{
        newNumbers := make([]int, 0)
        for _, number := range numbers{
                if fn(number) {
                        newNumbers = append(newNumbers, number)
                }
        }
        return newNumbers
}

func isEvenNumber(number int) bool{
        return number%2==0
}

func isOddNumber(number int) bool{
        return !isEvenNumber(number)
}

func GetOddNumbers(numbers []int) []int {
        return getNumbers(numbers, isOddNumber)
}

func GetEvenNumbers(numbers []int) []int {
        return getNumbers(numbers, isEvenNumber)
